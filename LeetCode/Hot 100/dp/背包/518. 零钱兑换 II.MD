    //完全背包问题，外循环nums,内循环target,target正序且target>=nums[i];
    class Solution {
    public int change(int amount, int[] coins) {
        int[] dp = new int[amount+1];
        Arrays.fill(dp,0);
        dp[0]=1;
        for(int coin : coins){
            for(int i =coin;i<=amount;i++){
                dp[i] +=  dp[i-coin];
            }
        }
        return dp[amount];
    }
    }
