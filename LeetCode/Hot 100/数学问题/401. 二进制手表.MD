class Solution {
    public List<String> readBinaryWatch(int turnedOn) {
        List<String> res = new ArrayList();
        int hour=0,min=0;
        for(int i=0;i<1024;i++){
            hour=i>>6;
            min=i&63;
            if(hour<12&&min<60&&Integer.bitCount(hour)+Integer.bitCount(min)==turnedOn){
                res.add(hour+":"+(min<10?"0":"")+min);
            }
        }
        return res;
    }
}
